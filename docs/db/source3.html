1. ST_MakePoint

SELECT ST_MakePoint(127.289231, 36.480057);
SELECT ST_MakePoint(127.2622522, 36.5024220);

2. ST_MakeLine

SELECT ST_MakeLine(ST_MakePoint(127.2615956, 36.5048945), ST_MakePoint(127.2603770, 36.5022889))

3. ST_GeomFromTEXT

SELECT ST_GeomFromText('POINT(127.289231 36.480057)', 4326);
SELECT ST_GeomFromText('LINESTRING(127.2615956 36.5048945, 127.2603770 36.5022889)', 4326);
SELECT ST_GeomFromText('POLYGON((127.2682235 36.4992861, 127.2688328 36.4990864, 127.2684956 36.4984017, 127.2678923 36.4986014, 127.2682235 36.4992861))', 4326);

4. ST_SRID

SELECT ST_SRID(location) FROM sejong_point
SELECT * FROM geometry_columns

5. ST_SetSRID

SELECT ST_SetSRID(ST_MakePoint(127.289231,36.480057), 4326);
SELECT ST_SetSRID(ST_MakeLine(ST_MakePoint(127.2615956, 36.5048945), ST_MakePoint(127.2603770, 36.5022889)), 4326);

6. ST_Transform

SELECT ST_Transform(location, 3857) FROM sejong_line;
SELECT ST_SRID(location) FROM sejong_line

10. ST_Length

11. ST_Distance

SELECT ST_Distance 
(ST_GeomFromText('Point(127.2595298 36.5019938)', 4326),

12. ST_DistanceSphere

SELECT ST_DistanceSphere 
(ST_GeomFromText('Point(127.2595298 36.5019938)', 4326),

13. ST_Intersect

SELECT * FROM yang_building as b, yang_flood as f
WHERE f.objectid_1='1' and ST_Intersects(b.geom, f.geom)

14. ST_Within

SELECT * FROM yang_building as b, yang_flood as f
WHERE f.objectid_1='1' and ST_Within(b.geom, f.geom)

15. ST_Overlap

SELECT * FROM yang_building as b, yang_flood as f
WHERE f.objectid_1='1' and ST_Overlap(b.geom, f.geom)

16. ST_Disjoint

SELECT * FROM yang_building AS b, yang_flood AS f
WHERE f.objectid_1='1' and ST_Disjoint(b.geom, f.geom)

17. ST_Buffer

SELECT * FROM yang_building as b, yang_flood as f
WHERE ST_INTERSECTS(b.geom, ST_Buffer(f.geom, 500))

18. ST_DWithin

SELECT * FROM yang_building as b, yang_flood as f
WHERE ST_DWithin(b.geom, f.geom, 500)
